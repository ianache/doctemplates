    ```python
    from sqlalchemy.orm import Session as SQLAlchemySession
    from app.models.user import User

    def upsert_user(db: SQLAlchemySession, sub: str, email: str) -> User:
        user = db.query(User).filter_by(sub=sub).first()
        if user is None:
            user = User(sub=sub, email=email)
            db.add(user)
        elif user.email != email:
            user.email = email
        db.commit()
        db.refresh(user)
        return user
    ```

    ```python
    import secrets
    from datetime import datetime, timedelta, timezone
    from sqlalchemy.orm import Session as SQLAlchemySession
    from app.config import settings
    from app.models.session import Session as SessionModel
    from app.models.user import User

    def create_session(db: SQLAlchemySession, user: User) -> SessionModel:
        session = SessionModel(
            id=secrets.token_urlsafe(32),
            user_id=user.id,
            expires_at=datetime.now(timezone.utc) + timedelta(seconds=settings.session_ttl_seconds),
        )
        db.add(session)
        db.commit()
        db.refresh(session)
        return session

    def delete_session(db: SQLAlchemySession, token: str) -> None:
        db.query(SessionModel).filter_by(id=token).delete()
        db.commit()
    ```
